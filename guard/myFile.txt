private void DoLoadRoutes(int Index) {
	NetHandler _handler = new NetHandler();//new NetHandler(MainLooper);
	_handler.mainThread = this;
	NetworkAccess na = new NetworkAccess(prgLayout);
	Map<String, String> p = new HashMap<String, String>();
	try {
		p.put("Token", Trans.Token);
		p.put("PageIndex", AES.aesEncrypt(Index + ""));
		Log.e(TAG, "Index:" + Index);
		Log.e(TAG, "AES.aesEncrypt(Index):" + AES.aesEncrypt(Index + ""));
		na.what = XGUARD.NETACTION.LOAD_ROUTES;
		na.execute(Trans.ServiceURL, "GetRoutes", _handler, p);
	} catch (Exception e) {
		Toast.makeText(thisView, e.getMessage(), Toast.LENGTH_LONG).show();
	}
}
